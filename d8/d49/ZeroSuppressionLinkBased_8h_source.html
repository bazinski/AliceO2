31 struct CommonHeader {
32  static constexpr uint32_t MagicWordLinkZS = 0xFC;
33  static constexpr uint32_t MagicWordTrigger = 0xAA;
34 
35  union {
36  uint64_t word0 = 0;
37  struct {
38  uint64_t bitMaskLow : 64;
39  };
40  };
41  union {
43  uint64_t word1 = 0;
44  struct {
45  uint64_t bitMaskHigh : 16;
46  uint32_t bunchCrossing : 12;
47  uint32_t numWordsPayload : 4;
48  uint32_t syncOffsetBC : 8;
49  uint32_t syncOffsetCRUCycles : 16;
50  uint32_t magicWord : 8;
51  };
52  };
53 
54  bool hasCorrectMagicWord() const { return (magicWord == MagicWordLinkZS) || (magicWord == MagicWordTrigger); }
55  bool isLinkZS() const { return (magicWord == MagicWordLinkZS); }
56  bool isTriggerInfo() const { return (magicWord == MagicWordTrigger); }
57 };

60 struct Header final : public CommonHeader {
61 
62  std::bitset<80> getChannelBits() const
63  {
64  return std::bitset<80>((std::bitset<80>(bitMaskHigh) << 64) | std::bitset<80>(bitMaskLow));
65  }
66 
67  bool isFillWord() const { return (word0 == 0xffffffffffffffff) && (word1 == 0xffffffffffffffff); }
68 };

80 template <uint32_t DataBitSizeT = 12, uint32_t SignificantBitsT = 2>
81 struct Data {
82  static constexpr uint32_t ChannelsPerWord = DataWordSizeBits / DataBitSizeT;
83  static constexpr uint32_t ChannelsPerHalfWord = ChannelsPerWord / 2;
84  static constexpr uint32_t DataBitSize = DataBitSizeT;
85  static constexpr uint32_t SignificantBits = SignificantBitsT;
86  static constexpr uint64_t BitMask = ((uint64_t(1) << DataBitSize) - 1);
87  static constexpr float FloatConversion = 1.f / float(1 << SignificantBits);
88 
89  uint64_t adcValues[2]{};
90 
92  void setADCValue(uint32_t pos, uint64_t value)
93  {
94  const uint32_t word = pos / ChannelsPerHalfWord;
95  const uint32_t posInWord = pos % ChannelsPerHalfWord;
96 
97  const uint64_t set = (value & BitMask) << (posInWord * DataBitSize);
98  const uint64_t mask = (0xFFFFFFFFFFFFFFFF ^ (BitMask << (posInWord * DataBitSize)));
99 
100  adcValues[word] &= mask;
101  adcValues[word] |= set;
102  }
103 
105  void setADCValueFloat(uint32_t pos, float value)
106  {
107  setADCValue(pos, uint64_t((value + 0.5f * FloatConversion) / FloatConversion));
108  }
109 
111  uint32_t getADCValue(uint32_t pos) const
112  {
113  const uint32_t word = pos / ChannelsPerHalfWord;
114  const uint32_t posInWord = pos % ChannelsPerHalfWord;
115 
116  return (adcValues[word] >> (posInWord * DataBitSize)) & BitMask;
117  }
118 
120  float getADCValueFloat(uint32_t pos) const
121  {
122  return float(getADCValue(pos)) * FloatConversion;
123  }
124 
126  void reset()
127  {
128  adcValues[0] = 0;
129  adcValues[1] = 0;
130  }
131 };

174 template <uint32_t DataBitSizeT = 12, uint32_t SignificantBitsT = 2, bool HasHeaderT = true>
175 struct Container {
176  static constexpr uint32_t ChannelsPerWord = DataWordSizeBits / DataBitSizeT;
177 
178  ContainerT<DataBitSizeT, SignificantBitsT, HasHeaderT> cont;
179 
181  uint32_t getADCValue(uint32_t word) { return cont.data[word / ChannelsPerWord].getADCValue(word % ChannelsPerWord); }
182 
184  float getADCValueFloat(uint32_t word) { return cont.data[word / ChannelsPerWord].getADCValueFloat(word % ChannelsPerWord); }
185 
187  void setADCValue(uint32_t word, uint64_t value) { cont.data[word / ChannelsPerWord].setADCValue(word % ChannelsPerWord, value); }
188 
190  void setADCValueFloat(uint32_t word, float value) { cont.data[word / ChannelsPerWord].setADCValueFloat(word % ChannelsPerWord, value); }
191 
193  void reset()
194  {
195  for (int i = 0; i < cont.getDataWords(); ++i) {
196  cont.data[i].reset();
197  }
198  }
199 
200  uint32_t getBunchCrossing() const { return cont.getBunchCrossing(); }
201  uint32_t getDataWords() const { return cont.getDataWords(); }
202  uint32_t getTotalWords() const { return cont.getTotalWords(); }
203  std::bitset<80> getChannelBits() const { return cont.getChannelBits(); }
204 
206  size_t getTotalSizeBytes() const { return getTotalWords() * DataWordSizeBytes; }
207 
209  Container* next() const
210  {
211  return (Container*)((const char*)this + getTotalWords() * DataWordSizeBytes);
212  }
